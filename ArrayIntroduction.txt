You'll be given an array of integers N and you have to print the integers in the reverse order.

SOLUTION:-

#include <cmath>
#include <cstdio>
#include <vector>
#include <iostream>
#include <algorithm>
using namespace std;


int main() {
    int N,i=0;
    cin>>N;
    int *A = new int[N];
    while(cin>>A[i++]);
    while(cout<<A[--N]<<' ' and N);
    delete[] A;

    
    return 0;
}
